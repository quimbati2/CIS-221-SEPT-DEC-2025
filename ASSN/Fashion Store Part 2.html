from flask import Flask, render_template_string import sqlite3 app = Flask(__name__) HTML_TEMPLATE = """ Fashion Store - Inventory

Fashion Store - Inventory Overview

{% for item in inventory %} {% endfor %}
Item Name Total Stock Sold In Stock Revenue ($)
{{ item[0] }} {{ item[1] }} {{ item[2] }} {{ item[3] }} {{ "%.2f"|format(item[4]) }}
""" def get_inventory_data(): conn = sqlite3.connect('database.db') cur = conn.cursor() cur.execute(""" SELECT item_name, total_stock, sold, (total_stock - sold) AS in_stock, (sold * price) AS revenue FROM inventory """) data = cur.fetchall() conn.close() return data @app.route("/") def index(): inventory = get_inventory_data() return render_template_string(HTML_TEMPLATE, inventory=inventory) if __name__ == "__main__": app.run(debug=True)
